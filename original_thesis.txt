Abstract

With the rapid development of Internet technology, Web frontend development has become one of the most popular technical directions in the software industry. However, traditional learning methods often suffer from scattered knowledge points, lack of practice environments, and unclear learning paths, leading to low learning efficiency and high dropout rates among beginners. To address these issues, this thesis designs and implements a frontend learning platform based on Vue3 and Spring Boot, aiming to provide an integrated and interactive online learning experience for frontend learners.

The platform adopts a B/S architecture with frontend-backend separation. The frontend is built as a responsive single-page application based on the Vue3 framework, combined with TypeScript, Pinia state management, Vue Router, Element Plus component library, TailwindCSS, and ECharts data visualization library. The backend provides stable and reliable RESTful API services using the Spring Boot framework, combined with Spring Data JPA, MySQL database, JWT authentication, and MinIO object storage. Additionally, the platform integrates the Coze AI API to implement an intelligent learning assistant and AI-powered exercise generation, providing personalized learning guidance for users.

The main features of the platform include: (1) progressive learning for seven frontend courses (HTML, CSS, JavaScript, Vue3, React, TypeScript, TailwindCSS) with step navigation and progress tracking; (2) AI-powered exercise generation and answer grading using edit distance algorithm for fuzzy matching; (3) wrong question notebook and mastered question management with retry and mastery marking; (4) learning assistance tools including course favorites, note-taking, course comments, and daily check-in; (5) global search functionality based on Fuse.js fuzzy search; (6) achievement system that automatically unlocks based on learning progress and check-in days; (7) admin management system with user management, role permissions, comment moderation, visit statistics, and data visualization.



Keywords: Vue3; Spring Boot; Frontend Learning Platform; AI-Assisted Teaching; Frontend-Backend Separation

function getEditDistance(a, b) {

  const m = a.length, n = b.length

  const dp = Array.from({ length: m + 1 }, () => Array(n + 1).fill(0))

  for (let i = 0; i <= m; i++) dp[i][0] = i

  for (let j = 0; j <= n; j++) dp[0][j] = j

  for (let i = 1; i <= m; i++)

    for (let j = 1; j <= n; j++)

      dp[i][j] = Math.min(

        dp[i-1][j] + 1,

        dp[i][j-1] + 1,

        dp[i-1][j-1] + (a[i-1] !== b[j-1] ? 1 : 0)

      )

  return dp[m][n]

}

 

function calculateSimilarity(a, b) {

  if (!a || !b) return 0

  const maxLen = Math.max(a.length, b.length)

  if (maxLen === 0) return 100

  return Math.round((1 - getEditDistance(a, b) / maxLen) * 100)

}

 

function gradeAnswer(standardAnswers, userAnswers) {

  let totalScore = 0

  for (const std of standardAnswers) {

    let bestSim = 0

    for (const usr of userAnswers) {

      bestSim = Math.max(bestSim, 

        calculateSimilarity(normalize(std), normalize(usr)))

    }

    totalScore += bestSim

  }

  return Math.round(totalScore / standardAnswers.length)

}

private int getConsecutiveDays(Integer userId) {

    List<CheckIn> list = checkInRepository

        .findByUserIdOrderByCheckInDateDesc(userId);

    if (list.isEmpty()) return 0;

    

    LocalDate today = LocalDate.now();

    if (!list.get(0).getCheckInDate().equals(today) 

        && !list.get(0).getCheckInDate().equals(today.minusDays(1)))

        return 0;

    

    int consecutive = 1;

    for (int i = 1; i < list.size(); i++) {

        LocalDate prev = list.get(i - 1).getCheckInDate();

        LocalDate curr = list.get(i).getCheckInDate();

        if (prev.equals(curr.plusDays(1))) consecutive++;

        else break;

    }

    return consecutive;

}

    }

    return getUnlocked(userId);
